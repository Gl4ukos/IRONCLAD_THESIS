<?xml version="1.0"?>
<robot xmlns:xacro="http://ros.org/wiki/xacro" name="ackermann_robot">

  <!-- Properties (all in meters) -->
  <xacro:property name="wheel_radius" value="0.035"/>
  <xacro:property name="wheel_width" value="0.03"/>
  <xacro:property name="chassis_length" value="0.35"/>
  <xacro:property name="chassis_width" value="0.14"/>
  <xacro:property name="chassis_height" value="0.075"/>
  <xacro:property name="wheelbase" value="0.23"/>
  <xacro:property name="steering_link_length" value="$(eval ${chassis_width}/2 - ${wheel_width}/2)"/>

  <!-- Base link -->
  <link name="base_link">
    <visual>
      <geometry>
        <box size="${chassis_length} ${chassis_width} ${chassis_height}"/>
      </geometry>
      <material name="blue"/>
    </visual>
    <collision>
      <geometry>
        <box size="${chassis_length} ${chassis_width} ${chassis_height}"/>
      </geometry>
    </collision>
    <inertial>
      <mass value="20"/>
      <origin xyz="0 0 0"/>
      <inertia ixx="1" ixy="0" ixz="0" iyy="1" iyz="0" izz="1"/>
    </inertial>
  </link>

  <!-- Rear wheel macro -->
  <xacro:macro name="rear_wheel" params="prefix side">
    <xacro:property name="x_pos" value="$(eval -${wheelbase}/2)"/>
    <xacro:property name="side_factor" value="$(eval 1 if '${side}'=='left' else -1)"/>
    <xacro:property name="y_pos" value="$(eval ${chassis_width}/2 * ${side_factor})"/>

    <link name="${prefix}_link">
      <visual>
        <geometry>
          <cylinder radius="${wheel_radius}" length="${wheel_width}"/>
        </geometry>
        <origin xyz="0 0 0" rpy="1.5708 0 0"/>
        <material name="black"/>
      </visual>
      <collision>
        <geometry>
          <cylinder radius="${wheel_radius}" length="${wheel_width}"/>
        </geometry>
        <origin xyz="0 0 0" rpy="1.5708 0 0"/>
      </collision>
      <inertial>
        <mass value="1"/>
        <origin xyz="0 0 0"/>
        <inertia ixx="0.01" ixy="0" ixz="0" iyy="0.01" iyz="0" izz="0.01"/>
      </inertial>
    </link>

    <joint name="${prefix}_joint" type="continuous">
      <parent link="base_link"/>
      <child link="${prefix}_link"/>
      <origin xyz="$(eval ${x_pos}) $(eval ${y_pos}) $(eval -${wheel_radius})" rpy="0 0 0"/>
      <axis xyz="0 1 0"/>
    </joint>
  </xacro:macro>

  <!-- Front steering wheel macro -->
  <xacro:macro name="front_wheel" params="prefix side">
    <xacro:property name="x_pos" value="$(eval ${wheelbase}/2)"/>
    <xacro:property name="side_factor" value="$(eval 1 if '${side}'=='left' else -1)"/>
    <xacro:property name="y_pos" value="$(eval ${chassis_width}/2 * ${side_factor})"/>

    <!-- Steering link -->
    <link name="${prefix}_steering_link">
      <visual>
        <geometry>
          <box size="${steering_link_length} 0.05 0.05"/>
        </geometry>
        <material name="gray"/>
      </visual>
      <collision>
        <geometry>
          <box size="${steering_link_length} 0.05 0.05"/>
        </geometry>
      </collision>
      <inertial>
        <mass value="1"/>
        <origin xyz="0 0 0"/>
        <inertia ixx="0.01" ixy="0" ixz="0" iyy="0.01" iyz="0" izz="0.01"/>
      </inertial>
    </link>

    <!-- Steering joint -->
    <joint name="${prefix}_steering_joint" type="continuous">
      <parent link="base_link"/>
      <child link="${prefix}_steering_link"/>
      <origin xyz="$(eval ${x_pos}) $(eval ${y_pos}) $(eval -${wheel_radius})" rpy="0 0 0"/>
      <axis xyz="0 0 1"/>
    </joint>

    <!-- Wheel link -->
    <link name="${prefix}_wheel_link">
      <visual>
        <geometry>
          <cylinder radius="${wheel_radius}" length="${wheel_width}"/>
        </geometry>
        <origin xyz="0 0 0" rpy="1.5708 0 0"/>
        <material name="black"/>
      </visual>
      <collision>
        <geometry>
          <cylinder radius="${wheel_radius}" length="${wheel_width}"/>
        </geometry>
        <origin xyz="0 0 0" rpy="1.5708 0 0"/>
      </collision>
      <inertial>
        <mass value="1"/>
        <origin xyz="0 0 0"/>
        <inertia ixx="0.01" ixy="0" ixz="0" iyy="0.01" iyz="0" izz="0.01"/>
      </inertial>
    </link>

    <!-- Wheel rotation joint -->
    <joint name="${prefix}_wheel_joint" type="continuous">
      <parent link="${prefix}_steering_link"/>
      <child link="${prefix}_wheel_link"/>
      <origin xyz="0 0 0"/>
      <axis xyz="0 1 0"/>
    </joint>
  </xacro:macro>

  <!-- Instantiate rear wheels -->
  <xacro:rear_wheel prefix="left_rear_wheel" side="left"/>
  <xacro:rear_wheel prefix="right_rear_wheel" side="right"/>

  <!-- Instantiate front wheels -->
  <xacro:front_wheel prefix="left_front" side="left"/>
  <xacro:front_wheel prefix="right_front" side="right"/>

  <!-- Transmissions -->
  <!-- Front steering -->
  <transmission name="left_front_steering_trans">
    <type>transmission_interface/SimpleTransmission</type>
    <joint name="left_front_steering_joint">
      <hardwareInterface>hardware_interface/PositionJointInterface</hardwareInterface>
    </joint>
    <actuator name="left_front_steering_motor">
      <mechanicalReduction>1</mechanicalReduction>
    </actuator>
  </transmission>

  <transmission name="right_front_steering_trans">
    <type>transmission_interface/SimpleTransmission</type>
    <joint name="right_front_steering_joint">
      <hardwareInterface>hardware_interface/PositionJointInterface</hardwareInterface>
    </joint>
    <actuator name="right_front_steering_motor">
      <mechanicalReduction>1</mechanicalReduction>
    </actuator>
  </transmission>

  <!-- Rear drive -->
  <transmission name="left_rear_wheel_trans">
    <type>transmission_interface/SimpleTransmission</type>
    <joint name="left_rear_wheel_joint">
      <hardwareInterface>hardware_interface/VelocityJointInterface</hardwareInterface>
    </joint>
    <actuator name="left_rear_wheel_motor">
      <mechanicalReduction>1</mechanicalReduction>
    </actuator>
  </transmission>

  <transmission name="right_rear_wheel_trans">
    <type>transmission_interface/SimpleTransmission</type>
    <joint name="right_rear_wheel_joint">
      <hardwareInterface>hardware_interface/VelocityJointInterface</hardwareInterface>
    </joint>
    <actuator name="right_rear_wheel_motor">
      <mechanicalReduction>1</mechanicalReduction>
    </actuator>
  </transmission>

  <!-- Gazebo plugins -->
  <gazebo>
    <plugin name="gazebo_ros_control" filename="libgazebo_ros_control.so">
      <robotNamespace>/ack</robotNamespace>
    </plugin>
  </gazebo>

  <gazebo>
    <plugin name="pose_publisher" filename="libgazebo_ros_p3d.so">
      <ros>
        <remapping>pose:=/base_link_pose</remapping>
      </ros>
      <bodyName>base_link</bodyName>
      <frameName>world</frameName>
      <updateRate>50</updateRate>
      <gaussianNoise>0.0</gaussianNoise>
      <alwaysOn>true</alwaysOn>
      <topicName>pose</topicName> 
    </plugin>
  </gazebo>

</robot>
